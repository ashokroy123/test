EMP_ID NUMBER PRIMARY KEY,

SALARY NUMBER(18,2) DEFAULT 0,
EXPERIENCE CHAR DEFAULT 'N',
MOBILE_NUMBER NUMBER NOT NULL,
EMAIL_ID VARCHAR(50) NOT NULL,
STATUS VARCHAR(10) DEFAULT 'ACTIVE'
);

select emp.EMP_ID, emp.FIRST_NAME, emp.LAST_NAME, emp.DATE_OF_JOIN, emp.DEPT_ID,dept.DEPT_NAME, emp.MOBILE_NUMBER, emp.EMAIL_ID, qual.QUALIFICATION_ID, qual.QUALIFICATION 
			from BSOL_EMPLOYEE_DETAILS emp 
			join BSOL_DEPARTMENT dept on dept.DEPT_ID = emp.DEPT_ID
			RIGHT outer join BSOL_EMPLOYEE_QUALIFICATIONS qual on qual.EMP_ID = emp.DEPT_ID
			where qual.IS_HIGHEST='Y';
